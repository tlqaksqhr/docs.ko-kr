## <a name="example"></a><span data-ttu-id="b9579-108">예</span><span class="sxs-lookup"><span data-stu-id="b9579-108">Example</span></span>

// cs1917.cs  
class CS1917  
{  
    public struct TestStruct  
    {  
        public int i;  
    }  
    public class C  
    {  
        public readonly TestStruct str = new TestStruct();  
        public static int Main()  
        {  
            C c = new C { str = { i = 1 } }; // CS1917  
            return 0;  
        }  
    }
