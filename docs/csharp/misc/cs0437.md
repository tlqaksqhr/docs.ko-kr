{  
   public class A {  
      public void Test() {

## <a name="example"></a><span data-ttu-id="2db36-108">예제</span><span class="sxs-lookup"><span data-stu-id="2db36-108">Example</span></span>

   public static void Main()

      Util.A x = new Util.A();

   }
